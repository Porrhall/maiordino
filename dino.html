<!DOCTYPE html>

<html>

<head>

    <style>

        #game {

            position: absolute;

            top: 0;

            left: 0;

        }


        #dinosaur {

            position: absolute;

            bottom: 0;

            left: 0;

        }

    </style>

</head>

<body>

    <canvas id="game" width="600" height="150"></canvas>

    <img id="dinosaur" src="dinosaur.png" alt="Dinosaur">

    <input type="file" id="customImage" accept="image/*">


    <script>

        const canvas = document.getElementById('game');

        const ctx = canvas.getContext('2d');

        const dinosaur = document.getElementById('dinosaur');

        const customImageInput = document.getElementById('customImage');


        let dinoX = 50;

        let dinoY = canvas.height - dinosaur.height;


        function draw() {

            ctx.clearRect(0, 0, canvas.width, canvas.height);

            ctx.drawImage(dinosaur, dinoX, dinoY);

        }


        function update() {

            draw();

        }


        function handleImageUpload(event) {

            const file = event.target.files[0];

            const reader = new FileReader();


            reader.onload = function (e) {

                dinosaur.src = e.target.result;

            };


            reader.readAsDataURL(file);

        }


        customImageInput.addEventListener('change', handleImageUpload);


        setInterval(update, 10);

    </script>

</body>

</html>
